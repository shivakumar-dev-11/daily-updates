switch Statement

Syntax:

switch (value) {
  case 'some case': {
    doThis;
    break;
  }

  case 'some other case': {
    doThis;
    break;
  }

  case 'nothing': {
    // don't do anything
    break;
  }

  default: {
    stayInGroundFloor;
  }
}
